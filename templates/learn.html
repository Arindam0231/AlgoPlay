Linear Regression

Step 1: Python Code

                
import numpy as np
from sklearn.linear_model import LinearRegression

x = np.array([2, 4, 6, 8]).reshape(-1, 1)
y = np.array([3, 7, 5, 10])

model = LinearRegression()
model.fit(x, y)

m = model.coef_[0]
b = model.intercept_

print(f"Slope (m): {m}")
print(f"Intercept (b): {b}")
print(f"Linear Equation: y = {m}x + {b}")
